\begin{frame}{Conclusão}
    \textbf{Contribuições principais}
    \begin{itemize}
        \item \textbf{Estradas promovem migração} -- resultado novo na literatura; efeito causal identificado via MST (instrumento geométrico).
        \item \textbf{Efeito sobre comércio} confirmado e quantificado: elasticidade VI $\approx -1{,}9$, superior ao MQO -- endogeneidade subestima o impacto das estradas.
        \item \textbf{Decomposição de bem-estar:} canais de comércio ($76\%$) e migração ($24\%$) separados dentro de um modelo estrutural coerente.
    \end{itemize}
\end{frame}
